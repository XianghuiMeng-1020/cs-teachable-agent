### Beginning of task description
Create a Python program that reads a text file named 'recipes.txt' and processes it to generate a report called 'ingredient_report.txt'. Each line in 'recipes.txt' contains a recipe formatted as: `recipe_name | ingredient1, ingredient2, ingredient3,...`. Your task is to generate a report with each ingredient followed by the number of times it appears across all recipes, sorted in descending order of frequency. Write the output to 'ingredient_report.txt', with each line formatted as: `ingredient: count`.

Your function should be named `generate_ingredient_report()`. This function does not take any parameters. Make sure to create this function and handle the file I/O within it. You will need to ignore lines in 'recipes.txt' that do not conform to the correct format of `recipe_name | ingredient1, ingredient2, ingredient3,...`. If 'recipes.txt' does not exist or is empty, the output file 'ingredient_report.txt' should also be empty.

Ensure that your solution handles different cases where ingredients may be unfortunately misspelled or written in different cases (e.g., 'Salt' vs 'salt') by considering each distinct form separately.

def test_generate_ingredient_report_general():
    generate_ingredient_report()
    with open(ingredient_report_file, 'r') as f:
        output = f.readlines()
    expected_output = [
        'salt: 5\n',
        'water: 3\n',
        'olive oil: 2\n',
        'carrots: 2\n',
        'flour: 2\n',
        'yeast: 2\n',
        'lettuce: 1\n',
        'beef: 1\n',
        'potatoes: 1\n',
        'garlic: 1\n'
    ]
    assert output == expected_output


def test_generate_ingredient_report_no_file():

    generate_ingredient_report()
    assert os.path.exists(ingredient_report_file)
    with open(ingredient_report_file, 'r') as f:
        output = f.read()
    assert output == ""


def test_generate_ingredient_report_empty_file():

    generate_ingredient_report()

        output = f.read()
    assert output == ""


def test_generate_ingredient_report_non_conforming_lines():

    with open(ingredient_report_file, 'r') as f:

    assert output == ""


def test_generate_ingredient_report_case_sensitivity():
    with open(recipes_file, 'w') as f:
        f.write("Pasta | Salt, olive oil, garlic\n")
        f.write("Salad | lettuce, OLIVE oil, salt\n")
    generate_ingredient_report()
    with open(ingredient_report_file, 'r') as f:
        output = f.readlines()
    expected_output = [
        'Salt: 1\n',
        'olive oil: 1\n',
        'garlic: 1\n',
        'lettuce: 1\n',
        'OLIVE oil: 1\n',
        'salt: 1\n'
    ]
    assert output == expected_output
